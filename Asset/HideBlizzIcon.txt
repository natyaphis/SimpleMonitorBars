SimpleMonitorBars - archived Blizzard tracker hide code

The following code was removed from the active addon because the related
checkbox was hidden and the suppression set was not wired to any actual
Blizzard tracker hiding behavior. The snippets are kept here for reference.

[Skins/MonitorTab.lua]
local hideTrackerCB = AceGUI:Create("CheckBox")
hideTrackerCB:SetLabel(L.mbHideFromTracker)
hideTrackerCB:SetValue(barCfg.hideFromCDM or false)
hideTrackerCB:SetCallback("OnValueChanged", function(_, _, val)
    barCfg.hideFromCDM = val
    MB:RebuildAllBars()
end)

[Bars/Bridge.lua]
ns.cdmSuppressedCooldownIDs = {}

function MB:ScanCDMViewers()
    if InCombatLockdown() then return end

    RefreshViewerCaches()

    self:PostScanHook()
    if self.RebuildCDMSuppressedSet then
        self:RebuildCDMSuppressedSet()
    end
end

[Bars/Bars.lua]
function MB:RebuildCDMSuppressedSet()
    -- Build the set of cooldown viewer entries hidden by user preference.
    local suppressed = ns.cdmSuppressedCooldownIDs
    wipe(suppressed)
    local bars = ns.db and ns.db.monitorBars and ns.db.monitorBars.bars
    if not bars then return end
    for _, barCfg in ipairs(bars) do
        if barCfg.enabled
            and IsClassMatchedForCurrentPlayer(barCfg.class)
            and barCfg.hideFromCDM
            and barCfg.spellID > 0 then
            local sid = barCfg.spellID
            local cdID = spellToCooldownID[sid]

            if not cdID and C_Spell and C_Spell.GetBaseSpell then
                local baseID = C_Spell.GetBaseSpell(sid)
                if baseID and baseID ~= sid then
                    cdID = spellToCooldownID[baseID]
                end
            end
            if cdID then
                suppressed[cdID] = true
            end
        end
    end
end
